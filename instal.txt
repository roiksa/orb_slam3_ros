Install CUDA and CUDNN

wget https://developer.download.nvidia.com/compute/cuda/repos/ubuntu2004/x86_64/cuda-ubuntu2004.pin
sudo mv cuda-ubuntu2004.pin /etc/apt/preferences.d/cuda-repository-pin-600

sudo add-apt-repository "deb http://developer.download.nvidia.com/compute/cuda/repos/ubuntu2004/x86_64/ /"

sudo apt update
sudo apt install cuda-toolkit-11-8

sudo apt install libcudnn8=*-1+cuda11.8
sudo apt-get install libcudnn8-dev=*-1+cuda11.8
sudo apt-get install libcudnn8-samples=*-1+cuda11.8



Install onnx runtime
cd ~
git clone https://github.com/microsoft/onnxruntime
cd onnxruntime
./build.sh --config Release --build_shared-lib --use_cuda --cuda_home /usr/local/cuda --cudnn_home /usr/lib/x86_64-linux-gnu/ --parallel
##extract include.zip ke onnxruntime/build/Linux/Release
